22 // ── Pinch / Zoom ────────────────────────────────────────────────────────
23 constexpr float PINCH_SCALE = 0.003f;
24 // Accumulated pinch-distance change (mm) that triggers one zoom step. Higher
25 // = LESS sensitive. 6 is a reliable middle ground; raise toward 8–10 if it
26 // feels twitchy, lower toward 4 if it feels sluggish.
27 constexpr float ZOOM_STEP_MM = 6.0f;
28 // Max one frame may add to the accumulator (mm). MUST be < ZOOM_STEP_MM so a
29 // single fast jump (e.g. the hand snapping open) can't fire a zoom alone.
30 constexpr float ZOOM_FRAME_CAP = 4.0f;
31 // A spread only counts as zoom-in while pinch is within this of its peak.
32 // Past that the pinch is easing = release starting, so outward motion is
33 // ignored. Small, so genuine pinch noise during a real zoom-in still passes.
34 constexpr float ZOOM_SPREAD_TOL = 0.03f;
35 // Max believable pinch-distance change in one frame (mm). Anything larger is
36 // a grab/release transient or glitch, not a deliberate zoom — ignored.
37 constexpr float PINCH_MAX_DELTA = 20.0f;
38 // Min pinch strength to zoom at all.
39 constexpr float ZOOM_PINCH_MIN = 0.85f;
40 // Per-frame DROP in pinch strength that means a fast snap-open (release).
41 // Kept high on purpose: a normal zoom-in spread also lowers pinch, just
42 // gradually, so only a sharp single-frame collapse should suppress zoom.
43 constexpr float ZOOM_RELEASE_DROP = 0.12f;
44 // HOLD-TO-CONTINUE zoom out: when pinch distance is squeezed below this (mm)
45 // — i.e. near the finger floor with no travel left — holding the tight pinch
46 // keeps zooming out, one step every ZOOM_CONTINUE_FRAMES frames.
47 constexpr float ZOOM_CONTINUE_TIGHT_MM = 7.0f;
48 // HOLD-TO-CONTINUE zoom in: when spread wider than this (mm) and the user
49 // actively spread to zoom in, holding the wide pinch keeps zooming in (the
50 // pose would otherwise cap zoom-in once you spread too far). Kept above a
51 // normal resting pinch so just holding a pinch doesn't drift into it.
52 constexpr float ZOOM_CONTINUE_WIDE_MM = 20.0f;
53 constexpr int ZOOM_CONTINUE_FRAMES = 20;
54 // How far pinch strength may fall below its peak before the rest of the pinch

56 // per-frame drop is too small to notice.
